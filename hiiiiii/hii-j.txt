import java.util.Scanner;
class Main{
    public static String getRevStr(String str) {
    int len = str.length();
    if (len < 2) {
        return str; // return the original string if it's too short
    }
    int mid = len / 2;
    char[] chars = str.toCharArray();
    for (int i = 0, j = mid; i < j; i++, j--) {
        char temp = chars[i];
        chars[i] = chars[j];
        chars[j] = temp;
    }
    return new String(chars);
}
    public static void main(String[] args)
    {
        Scanner sc=new Scanner(System.in);
        String st=sc.nextLine();
        String fr=getRevStr(st);
        System.out.println(fr);

class Main{
  public static void main(String[] args)
  {
    System.out.println("Enter any Number");
    Scanner sc=new Scanner(System.in);
    String st=sc.nextLine();
    st=st.toLowerCase();

    int count=0;

    String word[]=st.split("");
    for(int i=0; i<word.length; i++)
    {
      count=1;
      for(int j=i+1; j<word.length; j++)
      {
        if(word[i].equals(word[j]))
        {
          count++;
          word[j]="0";
        }

      }
      if(count>1 && word[i]!="0")
        System.out.print(word[i]);

    }
  }

class PrintOddEvenUsingTwoThread {
     int counter=1;
     static int n ;
    
public static void main(String[]args){
    n =10;
    PrintOddEvenUsingTwoThread poe = new  PrintOddEvenUsingTwoThread();
    Thread t1 = new Thread(new Runnable(){
        public void run(){
            poe.printOddNumber();
        }
    },"Thread-1");
    Thread t2 = new Thread(new Runnable(){
        public void run(){
            poe.printEvenNumber();
        }
    },"Thread-2");
    t1.start();
    t2.start();
}
public void printOddNumber(){
    while(counter<n){
        if(counter%2!=0){
            System.out.println(Thread.currentThread().getName()+" "+counter);
            counter++;
        }
    }
}
public void printEvenNumber(){
    while(counter<n){
        if(counter%2==0){
            System.out.println(Thread.currentThread().getName()+" "+counter);
            counter++;
    
        }
    }
}
}
